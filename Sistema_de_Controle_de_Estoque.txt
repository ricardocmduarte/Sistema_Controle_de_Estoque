<script>
//Entrada de dados e criação das variáveis saldo e encerrar:
saldo = parseInt(prompt("Informar saldo inicial"));
document.write("Saldo inicial: " + saldo + "<br>");
encerrar = "n";
//Estrutura de repetição while será repetido enquanto o valor da variável encerrar for "n": 
while (encerrar == "n")
{
//Instruções do while e criação das variáveis de entrada, tipo e quantidade:
tipo = parseInt(prompt("Informar tipo de entrada (digite: 1 (para compra de peças) ou digite: 2 (para venda de peças)"));
quantidade = parseInt(prompt("Informar quantidade"));
//Estrutura condicional para a entrada de dados e entrada no estoque do tipo 1: 
if(tipo == 1)
        {
        saldo = saldo + quantidade;
        document.write("Entrada: " + quantidade + "<br>");
        }
//Estrutura condicional para a entrada de dados e saída no estoque do tipo 2:
if(tipo == 2)
{
       if(saldo >= quantidade)       
               {
               saldo = saldo - quantidade;
               document.write("Saída: " + quantidade + "<br>");
               }
       else  
               {
               document.write("Saldo insuficiente" + "<br>");
               alert("Saldo insuficiente");
               }
}
document.write("Saldo: " + saldo + "<br>");
encerrar = prompt("Deseja encerrar ? (s/n)"); 
}
document.write("Sistema encerrado" + "<br>");
</script>
